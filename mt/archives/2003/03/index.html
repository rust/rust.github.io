この問題を適切に且つ迅速に解決するのが国連の役割であるはずなのに，個別 の利益にために互いの主張を繰り返すだけのものに成り下がっている印象を受 ける．世界はこのままどう動くのかを，冷静に見極める必要があるだろう．

$ su
# make install
  • PHP with apxs

--with-imap=../imap-2002 --with-imap-ssl=../openssl-0.9.7a
--enable-track-vars --enable-mbstr-enc-trans
--with-apxs2=/usr/local/apache2/bin/apxs
$ make
$ su
# make install

    $ ./configure --with-pgsql --enable-mbstring --with-mbregex
--with-imap=../imap-2002 --with-imap-ssl=../openssl-0.9.7a
--enable-track-vars --enable-mbstr-enc-trans
$ make
$ su
# make install
